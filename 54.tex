\begin{center}\section{ሱራቱ አልቀመር -  \textarabic{سوره  القمر}}\end{center}
\begin{longtable}{%

}
ቢስሚላሂ አራህመኒ ራሂይም &  \mytextarabic{بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ}\\
1.\  & \mytextarabic{ ٱقْتَرَبَتِ ٱلسَّاعَةُ وَٱنشَقَّ ٱلْقَمَرُ ﴿١﴾}\\
2.\  & \mytextarabic{وَإِن يَرَوْا۟ ءَايَةًۭ يُعْرِضُوا۟ وَيَقُولُوا۟ سِحْرٌۭ مُّسْتَمِرٌّۭ ﴿٢﴾}\\
3.\  & \mytextarabic{وَكَذَّبُوا۟ وَٱتَّبَعُوٓا۟ أَهْوَآءَهُمْ ۚ وَكُلُّ أَمْرٍۢ مُّسْتَقِرٌّۭ ﴿٣﴾}\\
4.\  & \mytextarabic{وَلَقَدْ جَآءَهُم مِّنَ ٱلْأَنۢبَآءِ مَا فِيهِ مُزْدَجَرٌ ﴿٤﴾}\\
5.\  & \mytextarabic{حِكْمَةٌۢ بَٰلِغَةٌۭ ۖ فَمَا تُغْنِ ٱلنُّذُرُ ﴿٥﴾}\\
6.\  & \mytextarabic{فَتَوَلَّ عَنْهُمْ ۘ يَوْمَ يَدْعُ ٱلدَّاعِ إِلَىٰ شَىْءٍۢ نُّكُرٍ ﴿٦﴾}\\
7.\  & \mytextarabic{خُشَّعًا أَبْصَـٰرُهُمْ يَخْرُجُونَ مِنَ ٱلْأَجْدَاثِ كَأَنَّهُمْ جَرَادٌۭ مُّنتَشِرٌۭ ﴿٧﴾}\\
8.\  & \mytextarabic{مُّهْطِعِينَ إِلَى ٱلدَّاعِ ۖ يَقُولُ ٱلْكَـٰفِرُونَ هَـٰذَا يَوْمٌ عَسِرٌۭ ﴿٨﴾}\\
9.\  & \mytextarabic{۞ كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٍۢ فَكَذَّبُوا۟ عَبْدَنَا وَقَالُوا۟ مَجْنُونٌۭ وَٱزْدُجِرَ ﴿٩﴾}\\
10.\  & \mytextarabic{فَدَعَا رَبَّهُۥٓ أَنِّى مَغْلُوبٌۭ فَٱنتَصِرْ ﴿١٠﴾}\\
11.\  & \mytextarabic{فَفَتَحْنَآ أَبْوَٟبَ ٱلسَّمَآءِ بِمَآءٍۢ مُّنْهَمِرٍۢ ﴿١١﴾}\\
12.\  & \mytextarabic{وَفَجَّرْنَا ٱلْأَرْضَ عُيُونًۭا فَٱلْتَقَى ٱلْمَآءُ عَلَىٰٓ أَمْرٍۢ قَدْ قُدِرَ ﴿١٢﴾}\\
13.\  & \mytextarabic{وَحَمَلْنَـٰهُ عَلَىٰ ذَاتِ أَلْوَٟحٍۢ وَدُسُرٍۢ ﴿١٣﴾}\\
14.\  & \mytextarabic{تَجْرِى بِأَعْيُنِنَا جَزَآءًۭ لِّمَن كَانَ كُفِرَ ﴿١٤﴾}\\
15.\  & \mytextarabic{وَلَقَد تَّرَكْنَـٰهَآ ءَايَةًۭ فَهَلْ مِن مُّدَّكِرٍۢ ﴿١٥﴾}\\
16.\  & \mytextarabic{فَكَيْفَ كَانَ عَذَابِى وَنُذُرِ ﴿١٦﴾}\\
17.\  & \mytextarabic{وَلَقَدْ يَسَّرْنَا ٱلْقُرْءَانَ لِلذِّكْرِ فَهَلْ مِن مُّدَّكِرٍۢ ﴿١٧﴾}\\
18.\  & \mytextarabic{كَذَّبَتْ عَادٌۭ فَكَيْفَ كَانَ عَذَابِى وَنُذُرِ ﴿١٨﴾}\\
19.\  & \mytextarabic{إِنَّآ أَرْسَلْنَا عَلَيْهِمْ رِيحًۭا صَرْصَرًۭا فِى يَوْمِ نَحْسٍۢ مُّسْتَمِرٍّۢ ﴿١٩﴾}\\
20.\  & \mytextarabic{تَنزِعُ ٱلنَّاسَ كَأَنَّهُمْ أَعْجَازُ نَخْلٍۢ مُّنقَعِرٍۢ ﴿٢٠﴾}\\
21.\  & \mytextarabic{فَكَيْفَ كَانَ عَذَابِى وَنُذُرِ ﴿٢١﴾}\\
22.\  & \mytextarabic{وَلَقَدْ يَسَّرْنَا ٱلْقُرْءَانَ لِلذِّكْرِ فَهَلْ مِن مُّدَّكِرٍۢ ﴿٢٢﴾}\\
23.\  & \mytextarabic{كَذَّبَتْ ثَمُودُ بِٱلنُّذُرِ ﴿٢٣﴾}\\
24.\  & \mytextarabic{فَقَالُوٓا۟ أَبَشَرًۭا مِّنَّا وَٟحِدًۭا نَّتَّبِعُهُۥٓ إِنَّآ إِذًۭا لَّفِى ضَلَـٰلٍۢ وَسُعُرٍ ﴿٢٤﴾}\\
25.\  & \mytextarabic{أَءُلْقِىَ ٱلذِّكْرُ عَلَيْهِ مِنۢ بَيْنِنَا بَلْ هُوَ كَذَّابٌ أَشِرٌۭ ﴿٢٥﴾}\\
26.\  & \mytextarabic{سَيَعْلَمُونَ غَدًۭا مَّنِ ٱلْكَذَّابُ ٱلْأَشِرُ ﴿٢٦﴾}\\
27.\  & \mytextarabic{إِنَّا مُرْسِلُوا۟ ٱلنَّاقَةِ فِتْنَةًۭ لَّهُمْ فَٱرْتَقِبْهُمْ وَٱصْطَبِرْ ﴿٢٧﴾}\\
28.\  & \mytextarabic{وَنَبِّئْهُمْ أَنَّ ٱلْمَآءَ قِسْمَةٌۢ بَيْنَهُمْ ۖ كُلُّ شِرْبٍۢ مُّحْتَضَرٌۭ ﴿٢٨﴾}\\
29.\  & \mytextarabic{فَنَادَوْا۟ صَاحِبَهُمْ فَتَعَاطَىٰ فَعَقَرَ ﴿٢٩﴾}\\
30.\  & \mytextarabic{فَكَيْفَ كَانَ عَذَابِى وَنُذُرِ ﴿٣٠﴾}\\
31.\  & \mytextarabic{إِنَّآ أَرْسَلْنَا عَلَيْهِمْ صَيْحَةًۭ وَٟحِدَةًۭ فَكَانُوا۟ كَهَشِيمِ ٱلْمُحْتَظِرِ ﴿٣١﴾}\\
32.\  & \mytextarabic{وَلَقَدْ يَسَّرْنَا ٱلْقُرْءَانَ لِلذِّكْرِ فَهَلْ مِن مُّدَّكِرٍۢ ﴿٣٢﴾}\\
33.\  & \mytextarabic{كَذَّبَتْ قَوْمُ لُوطٍۭ بِٱلنُّذُرِ ﴿٣٣﴾}\\
34.\  & \mytextarabic{إِنَّآ أَرْسَلْنَا عَلَيْهِمْ حَاصِبًا إِلَّآ ءَالَ لُوطٍۢ ۖ نَّجَّيْنَـٰهُم بِسَحَرٍۢ ﴿٣٤﴾}\\
35.\  & \mytextarabic{نِّعْمَةًۭ مِّنْ عِندِنَا ۚ كَذَٟلِكَ نَجْزِى مَن شَكَرَ ﴿٣٥﴾}\\
36.\  & \mytextarabic{وَلَقَدْ أَنذَرَهُم بَطْشَتَنَا فَتَمَارَوْا۟ بِٱلنُّذُرِ ﴿٣٦﴾}\\
37.\  & \mytextarabic{وَلَقَدْ رَٰوَدُوهُ عَن ضَيْفِهِۦ فَطَمَسْنَآ أَعْيُنَهُمْ فَذُوقُوا۟ عَذَابِى وَنُذُرِ ﴿٣٧﴾}\\
38.\  & \mytextarabic{وَلَقَدْ صَبَّحَهُم بُكْرَةً عَذَابٌۭ مُّسْتَقِرٌّۭ ﴿٣٨﴾}\\
39.\  & \mytextarabic{فَذُوقُوا۟ عَذَابِى وَنُذُرِ ﴿٣٩﴾}\\
40.\  & \mytextarabic{وَلَقَدْ يَسَّرْنَا ٱلْقُرْءَانَ لِلذِّكْرِ فَهَلْ مِن مُّدَّكِرٍۢ ﴿٤٠﴾}\\
41.\  & \mytextarabic{وَلَقَدْ جَآءَ ءَالَ فِرْعَوْنَ ٱلنُّذُرُ ﴿٤١﴾}\\
42.\  & \mytextarabic{كَذَّبُوا۟ بِـَٔايَـٰتِنَا كُلِّهَا فَأَخَذْنَـٰهُمْ أَخْذَ عَزِيزٍۢ مُّقْتَدِرٍ ﴿٤٢﴾}\\
43.\  & \mytextarabic{أَكُفَّارُكُمْ خَيْرٌۭ مِّنْ أُو۟لَـٰٓئِكُمْ أَمْ لَكُم بَرَآءَةٌۭ فِى ٱلزُّبُرِ ﴿٤٣﴾}\\
44.\  & \mytextarabic{أَمْ يَقُولُونَ نَحْنُ جَمِيعٌۭ مُّنتَصِرٌۭ ﴿٤٤﴾}\\
45.\  & \mytextarabic{سَيُهْزَمُ ٱلْجَمْعُ وَيُوَلُّونَ ٱلدُّبُرَ ﴿٤٥﴾}\\
46.\  & \mytextarabic{بَلِ ٱلسَّاعَةُ مَوْعِدُهُمْ وَٱلسَّاعَةُ أَدْهَىٰ وَأَمَرُّ ﴿٤٦﴾}\\
47.\  & \mytextarabic{إِنَّ ٱلْمُجْرِمِينَ فِى ضَلَـٰلٍۢ وَسُعُرٍۢ ﴿٤٧﴾}\\
48.\  & \mytextarabic{يَوْمَ يُسْحَبُونَ فِى ٱلنَّارِ عَلَىٰ وُجُوهِهِمْ ذُوقُوا۟ مَسَّ سَقَرَ ﴿٤٨﴾}\\
49.\  & \mytextarabic{إِنَّا كُلَّ شَىْءٍ خَلَقْنَـٰهُ بِقَدَرٍۢ ﴿٤٩﴾}\\
50.\  & \mytextarabic{وَمَآ أَمْرُنَآ إِلَّا وَٟحِدَةٌۭ كَلَمْحٍۭ بِٱلْبَصَرِ ﴿٥٠﴾}\\
51.\  & \mytextarabic{وَلَقَدْ أَهْلَكْنَآ أَشْيَاعَكُمْ فَهَلْ مِن مُّدَّكِرٍۢ ﴿٥١﴾}\\
52.\  & \mytextarabic{وَكُلُّ شَىْءٍۢ فَعَلُوهُ فِى ٱلزُّبُرِ ﴿٥٢﴾}\\
53.\  & \mytextarabic{وَكُلُّ صَغِيرٍۢ وَكَبِيرٍۢ مُّسْتَطَرٌ ﴿٥٣﴾}\\
54.\  & \mytextarabic{إِنَّ ٱلْمُتَّقِينَ فِى جَنَّـٰتٍۢ وَنَهَرٍۢ ﴿٥٤﴾}\\
55.\  & \mytextarabic{فِى مَقْعَدِ صِدْقٍ عِندَ مَلِيكٍۢ مُّقْتَدِرٍۭ ﴿٥٥﴾}\\
\end{longtable}
